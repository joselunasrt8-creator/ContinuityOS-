        id: constitutional_drift
        shell: bash

          set -euo pipefail

          CONSTITUTIONAL_PATHS=(
            "governance/"
            "src/runtime/validate.ts"
            "src/runtime/execute.ts"

            "src/runtime/authority.ts"

          cat changed_files.txt

name: standing-authority-issuance
# owner: #1831
# policy: GAP-005 — Policy-Bound GMA (Standing Authority, GMA Phase 2)
# policy-path: governance/authorizations/STANDING_AUTHORITY_SPEC.json
# gap-reference: GAP-005
#
# Issues a Standing Authority (SA): a bounded, owner-issued authority from which
# scoped governance-mutation authorizations are DERIVED automatically at merge-check
# time (Tier 3 of merge-governance-check.yml), instead of a manual per-PR GMA.
#
# One issuance -> many bounded derivations. The compressed rule:
#   Manual GMA governs one mutation.
#   Standing Authority governs a bounded class of mutations.
#   Both must end in proof.
#
# Bound model: TTL = hard bound; max_merges = async proof-ledger budget.
# V1 issues NARROW by default (one mutation class / one surface). Widen later.
#
# TWO-STEP ISSUANCE (GAP-005 Phase 2, gate-both): standing_authority_registry.jsonl
# is now a governed trust surface (governance_mutation), so the issuance PR ITSELF
# requires a GMA and cannot self-derive (trust surfaces are hard-denied from Tier 3).
# To land a Standing Authority on a protected branch:
#   1. Run THIS workflow on a branch  -> appends the SA to standing_authority_registry.jsonl
#   2. Run governance-mutation-authorization.yml on the SAME branch
#                                      -> emits a GMA covering the SA-registry change
#   3. Open the PR and merge           -> Tier 1 (explicit GMA) authorizes the issuance
# Both steps are owner workflow_dispatch, so issuance retains owner provenance.

on:
  workflow_dispatch:
    inputs:
      branch_pattern:
        description: 'Branch glob the authority applies to (e.g. claude/*)'
        required: true
        default: 'claude/*'
      mutation_classes:
        description: 'Comma-separated allowed mutation classes (e.g. workflow_mutation)'
        required: true
        default: 'workflow_mutation'
      path_globs:
        description: 'Comma-separated allowed path globs (e.g. .github/workflows/**)'
        required: true
        default: '.github/workflows/**'
      ttl_hours:
        description: 'Time-to-live in hours (hard bound)'
        required: false
        default: '24'
      max_merges:
        description: 'Async proof-ledger budget — max governed merges under this authority'
        required: false
        default: '2'
      intent:
        description: 'Human-readable description of the bounded authority'
        required: true

permissions:
  contents: write

jobs:
  issue-standing-authority:
    if: ${{ !startsWith(github.ref_name, 'proof-registry/') }}
    runs-on: ubuntu-latest

      PATH_GLOBS: ${{ inputs.path_globs }}
